  def make_request(method, path)
    require 'net/http'
    require 'net/https'

    url = URI.parse("#{gemcutter_url}/#{path}")

    http = proxy_class.new(url.host, url.port)

    if url.scheme == 'https'
      http.verify_mode = OpenSSL::SSL::VERIFY_NONE
      http.use_ssl = true
    end

    request_method =
      case method
      when :get
        proxy_class::Get
      when :post
        proxy_class::Post
      when :put
        proxy_class::Put
      when :delete
        proxy_class::Delete
      else
        raise ArgumentError
      end

    request = request_method.new(url.path)
    yield request if block_given?
    http.request(request)
  end
