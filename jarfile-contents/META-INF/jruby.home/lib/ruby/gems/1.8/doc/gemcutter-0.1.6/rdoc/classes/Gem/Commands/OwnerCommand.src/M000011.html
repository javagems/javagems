  def initialize
    super 'owner', description
    defaults.merge!(:add => [], :remove => [])

    add_option('-a', '--add EMAIL', 'Add an owner') do |value, options|
      options[:add] << value
    end

    add_option('-r', '--remove EMAIL', 'Remove an owner') do |value, options|
      options[:remove] << value
    end
  end
