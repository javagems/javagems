    def initialize(proxy = nil)
      @debug_dev       = nil
      @ssl_verify_mode = OpenSSL::SSL::VERIFY_NONE
      if proxy
        begin
          proxy_uri = URI.parse(proxy)
          @agent_class = Net::HTTP::Proxy(proxy_uri.host,proxy_uri.port)
        rescue URI::InvalidURIError
        end
      end
      @agent_class ||= Net::HTTP
    end
