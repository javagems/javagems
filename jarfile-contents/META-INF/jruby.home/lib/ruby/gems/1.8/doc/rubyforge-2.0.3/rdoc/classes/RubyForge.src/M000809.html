  def post_news(group_id, subject, body)
    # TODO - what was the post_changes parameter for?
    form = {
      "news_byte[summary]"      => subject,
      "news_byte[details]"      => body
    }
    group_id = lookup "group", group_id
    url = "/groups/#{group_id}/news_bytes"
    run url, form
  end
