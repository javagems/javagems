  def sign_in
    say "Enter your #{DESTINATION_NAME} credentials. Don't have an account yet? Create one at #{URL}/sign_up"

    email = ask("Email: ")
    password = ask_for_password("Password: ")

    response = make_request(:get, "api_key") do |request|
      request.basic_auth email, password
    end

    case response
    when Net::HTTPSuccess
      Gem.configuration[:gemcutter_key] = response.body
      Gem.configuration.write
      say "Signed in. Your api key has been stored in #{Gem.configuration.config_file_name}"
    else
      say response.body
      terminate_interaction
    end
  end
